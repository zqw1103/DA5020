# Q4

library(ggplot2)
ggplot(mtcars,aes(x=disp, y=mpg))


##no chart only background and x,y , missing geom_jitter or geom_point layer to produce scatterplot. Generally speaking, geometric object needs to be included and specified appropriately
ggplot(mtcars,aes(x=disp, y=mpg)) + geom_jitter()


# Q5

ggplot(mpg,aes(x=displ, y=hwy)) + geom_jitter(aes(color=cyl))


#overall trend is linear relationship between hwy and displ variables with higher hwy lower displ. According to cyl, it separates data into 3 sections with exceptions of some cyl=8 cars where it has much higher hwy compared with same cyl cars based on same displ.


# Q6

ggplot(mpg,aes(x=displ, y=hwy)) + geom_jitter(aes(color=cyl)) +
  facet_wrap(~drv)


# Rear wheel drive vehicles tend to be low in hwy due to the majority of cyl=8 cars with large displ while front wheel drive appears to have higher hwy with low displ with smaller cyl and 4 wheel drive cars is more spread across displ with more cyl possibilities.
